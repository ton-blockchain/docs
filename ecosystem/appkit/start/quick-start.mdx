Use this page to connect a wallet from a React application with the smallest useful setup.

## Before you begin

You need:

- a browser application;
- a public TON Connect manifest URL;
- an RPC provider URL;
- AppKit packages installed.

## Create AppKit

```ts
import { AppKit, Network, createTonConnectConnector } from '@ton/appkit';

export const appKit = new AppKit({
  networks: {
    [Network.testnet().chainId]: {
      apiClient: { url: 'https://testnet.toncenter.com' },
    },
  },
  connectors: [
    createTonConnectConnector({
      tonConnectOptions: {
        manifestUrl: 'https://example.com/tonconnect-manifest.json',
      },
    }),
  ],
});
```

## Add providers

```tsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { AppKitProvider } from '@ton/appkit-react';
import '@ton/appkit-react/styles.css';
import { appKit } from './appkit';

const queryClient = new QueryClient();

export function Root() {
  return (
    <QueryClientProvider client={queryClient}>
      <AppKitProvider appKit={appKit}>
        <App />
      </AppKitProvider>
    </QueryClientProvider>
  );
}
```

## Connect a wallet

```tsx
import { useConnect, useConnectors, useSelectedWallet } from '@ton/appkit-react';

export function WalletPanel() {
  const { data: connectors = [] } = useConnectors();
  const connect = useConnect();
  const { data: wallet } = useSelectedWallet();

  if (wallet) {
    return <div>Connected: {wallet.address}</div>;
  }

  return connectors.map((connector) => (
    <button key={connector.id} onClick={() => connect.mutate({ connector })}>
      Connect {connector.name}
    </button>
  ));
}
```
